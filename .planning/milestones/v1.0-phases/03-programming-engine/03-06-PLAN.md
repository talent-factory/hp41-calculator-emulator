pub fn dispatch(state: &mut CalcState, op: Op) -> Result<(), HpError> {
    flush_entry_buf(state)?;

    // ── Phase 3: PRGM mode recording gate ──────────────────────────────
    if state.prgm_mode {
        if matches!(op, Op::PrgmMode) {
            state.prgm_mode = false;
            apply_lift_effect(state, LiftEffect::Neutral);
            return Ok(());
        }
        state.program.push(op);
        return Ok(());
    }

    match op {
        Op::Add => op_add(state),
        // ... existing Phase 1+2 arms ...
        Op::AlphaToggle     => op_alpha_toggle(state),
        Op::AlphaAppend(ch) => op_alpha_append(state, ch),
        Op::AlphaClear      => op_alpha_clear(state),
        // Phase 3 arms NOT YET WIRED (catch-all currently here)
        _ => Err(HpError::InvalidOp),  // <-- replace this with specific arms
    }
}

        // Note: PrgmMode exit is handled by the prgm_mode gate above (never reaches here).

        Op::Gto(s)     => program::op_gto(state, &s),

            program::op_isg(state, reg).map(|_| ())

        Op::Dse(reg)   => {
            program::op_dse(state, reg).map(|_| ())
