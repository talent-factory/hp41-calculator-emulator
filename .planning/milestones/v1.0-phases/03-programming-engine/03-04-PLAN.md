```rust
// flush_entry_buf replacement (after prgm_mode field exists on state):
pub fn flush_entry_buf(state: &mut CalcState) -> Result<(), HpError> {
    if state.entry_buf.is_empty() { return Ok(()); }
    let s = state.entry_buf.clone();
    state.entry_buf.clear();
    let d = Decimal::from_str(&s).map_err(|_| HpError::InvalidOp)?;
    let n = HpNum::rounded(d);
    if state.prgm_mode {
        state.program.push(Op::PushNum(n));
        // lift_enabled unchanged in recording mode — recording does not affect execution state
    } else {
        crate::stack::enter_number(state, n);
        crate::stack::apply_lift_effect(state, LiftEffect::Enable);
    }
    Ok(())
}

// prgm_mode gate — insert in dispatch() after flush_entry_buf(state)?, before match op:
if state.prgm_mode {
    if matches!(op, Op::PrgmMode) {
        // PrgmMode op while recording = exit recording immediately (HP-41 toggle, D-03/Pitfall 6)
        state.prgm_mode = false;
        apply_lift_effect(state, LiftEffect::Neutral);
        return Ok(());
    }
    // All other ops: record to program Vec, do NOT execute
    state.program.push(op);
    return Ok(());
}
```
