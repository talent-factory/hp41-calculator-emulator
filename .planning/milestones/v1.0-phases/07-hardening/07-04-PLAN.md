    ```rust
    #[cfg(test)]
    #[allow(clippy::unwrap_used)]
    mod program_tests {
        use crate::error::HpError;
        use crate::num::HpNum;
        use crate::ops::{dispatch, Op, TestKind};
        use crate::ops::program::{op_gto, op_xeq, op_rtn, op_isg, op_dse, parse_counter};
        use crate::state::CalcState;
        use rust_decimal::Decimal;
        use std::str::FromStr;

        // Helper: build a CalcState with a program already loaded
        fn state_with_program(ops: Vec<Op>) -> CalcState {
            let mut s = CalcState::default();
            s.program = ops;
            s
        }

        #[test]
        fn test_run_program_label_not_found() {
            let mut state = CalcState::default();
            // Empty program: no label "A" exists
            let result = crate::ops::program::run_program(&mut state, "A");
            assert_eq!(result, Err(HpError::InvalidOp));
        }

        #[test]
        fn test_run_program_is_running_reset_on_error() {
            let mut state = CalcState::default();
            let _ = crate::ops::program::run_program(&mut state, "A"); // triggers error
            // is_running must be reset to false even on error (safety reset pattern)
            assert!(!state.is_running, "is_running must be false after run_program error");
        }

        #[test]
        fn test_call_depth_limit() {
            // Build 6 labels A–F where each calls the next, 5 levels deep
            // A calls B (depth 1), B calls C (depth 2), C calls D (depth 3),
            // D calls E (depth 4 = limit), E calls F (depth 5 = exceeds limit)
            let program = vec![
                Op::Lbl("A".to_string()),
                Op::Xeq("B".to_string()),
                Op::Rtn,
                Op::Lbl("B".to_string()),
                Op::Xeq("C".to_string()),
                Op::Rtn,
                Op::Lbl("C".to_string()),
                Op::Xeq("D".to_string()),
                Op::Rtn,
                Op::Lbl("D".to_string()),
                Op::Xeq("E".to_string()),
                Op::Rtn,
                Op::Lbl("E".to_string()),
                Op::Xeq("F".to_string()), // 5th nested XEQ — exceeds 4-level limit
                Op::Rtn,
                Op::Lbl("F".to_string()),
                Op::Rtn,
            ];
            let mut state = state_with_program(program);
            let result = crate::ops::program::run_program(&mut state, "A");
            assert_eq!(result, Err(HpError::CallDepth),
                "5th nested XEQ must return CallDepth error");
        }

        #[test]
        fn test_max_steps_infinite_loop_guard() {
            // Infinite loop: Lbl A → Gto A → ... (runs forever without guard)
            let program = vec![
                Op::Lbl("A".to_string()),
                Op::Gto("A".to_string()),
            ];
            let mut state = state_with_program(program);
            let result = crate::ops::program::run_program(&mut state, "A");
            // MAX_STEPS = 1_000_000; guard returns HpError::Overflow
            assert_eq!(result, Err(HpError::Overflow),
                "Infinite loop must be caught by MAX_STEPS guard");
        }

        #[test]
        fn test_op_isg_reg_out_of_bounds() {
            let mut state = CalcState::default();
            // Default regs has 100 entries (0–99); reg 100 is out of bounds
            let result = op_isg(&mut state, 100);
            assert_eq!(result, Err(HpError::InvalidOp));
        }

        #[test]
        fn test_op_dse_reg_out_of_bounds() {
            let mut state = CalcState::default();
            let result = op_dse(&mut state, 100);
            assert_eq!(result, Err(HpError::InvalidOp));
        }

        #[test]
        fn test_op_gto_interactive_invalid() {
            let mut state = CalcState::default();
            state.is_running = false;
            let result = op_gto(&mut state, "A");
            assert_eq!(result, Err(HpError::InvalidOp));
        }

        #[test]
        fn test_op_xeq_interactive_invalid() {
            let mut state = CalcState::default();
            state.is_running = false;
            let result = op_xeq(&mut state, "A");
            assert_eq!(result, Err(HpError::InvalidOp));
        }

        #[test]
        fn test_gto_label_not_found_during_run() {
            // A program that jumps to a non-existent label during execution
            let program = vec![
                Op::Lbl("A".to_string()),
                Op::Gto("MISSING".to_string()),
            ];
            let mut state = state_with_program(program);
            let result = crate::ops::program::run_program(&mut state, "A");
            assert_eq!(result, Err(HpError::InvalidOp));
        }

        #[test]
        fn test_parse_counter_canonical_phase3_example() {
            // Phase 3 canonical: counter 1.00500 → current=1, final=5, step=1
            // Note: rust_decimal normalises 1.00500 → "1.005" (trailing zero strip)
            let n = HpNum(Decimal::from_str("1.005").unwrap());
            let (current, final_val, step, frac_padded) = parse_counter(&n).unwrap();
            assert_eq!(current, 1, "current must be 1");
            assert_eq!(final_val, 5, "final must be 5 (from '005')");
            assert_eq!(step, 1, "step '00' must default to 1");
            assert_eq!(&frac_padded, "00500",
                "frac_padded must be right-padded to 5 chars: '00500'");
        }

        #[test]
        fn test_isg_increments_and_then_skips() {
            // counter 4.00500 → current=4, final=5, step=1
            // After 1st isg: current=5, 5 > 5 = false → loop continues (Ok(false))
            // After 2nd isg: current=6, 6 > 5 = true → loop exits (Ok(true))
            let mut state = CalcState::default();
            state.regs[0] = HpNum(Decimal::from_str("4.005").unwrap());
            let result1 = op_isg(&mut state, 0).unwrap();
            assert!(!result1, "isg at current=4 (new=5 not > final=5): must NOT skip");
            let result2 = op_isg(&mut state, 0).unwrap();
            assert!(result2, "isg at current=5 (new=6 > final=5): must skip (exit loop)");
        }

        #[test]
        fn test_rtn_interactive_noop() {
            // Interactive RTN when not running and call_stack is empty = no-op
            let mut state = CalcState::default();
            state.is_running = false;
            assert!(state.call_stack.is_empty());
            let result = op_rtn(&mut state);
            assert!(result.is_ok(), "op_rtn with empty call_stack must return Ok(())");
            assert!(state.call_stack.is_empty(), "call_stack must remain empty");
        }
    }
    ```
