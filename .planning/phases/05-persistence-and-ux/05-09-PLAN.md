
  After the fix the sequence must be exactly:
  ```rust
        Op::PushNum(HpNum::from(2i32)),
        Op::RclReg(0),
        Op::Test(TestKind::XLeY),           // if n <= 2: prime
        Op::Gto("P".to_string()),
  ```

  With this fix: stack at Test time is X=n, Y=2 → XLeY checks n≤Y i.e. n≤2, which correctly
  routes n=2 (and n<2) to the prime label P.

  ```rust
      // Mean of 4 values already on the stack (T, Z, Y, X).
      // Adds all four: T+Z+Y+X, then divides by 4 → mean in X.
      // No registers used. No indirect addressing needed.
  ```

  ```rust
  fn test_prime_test_correctness() {

      let run_prime = |n: i32| -> f64 {
          let mut state = CalcState::new();
          state.program = prime_prog.ops.clone();
          hp41_core::ops::dispatch(&mut state, Op::PushNum(HpNum::from(n))).unwrap();
          hp41_core::run_program(&mut state, "A")
              .unwrap_or_else(|e| panic!("prime({n}) failed: {e:?}"));
          // HpNum -> f64 via Into<f64> or similar; use the same pattern as HpNum::from
          f64::from(state.stack.x)

      assert_eq!(run_prime(4), 0.0, "prime(4) must be 0 (composite)");
      assert_eq!(run_prime(9), 0.0, "prime(9) must be 0 (composite: 3×3)");
      assert_eq!(run_prime(13), 1.0, "prime(13) must be 1 (prime)");

      state.program = mean_prog.ops.clone();
      // Push 4 values: 1, 2, 3, 4 → T=1, Z=2, Y=3, X=4
      for n in [1i32, 2, 3, 4] {
          hp41_core::ops::dispatch(&mut state, Op::PushNum(HpNum::from(n))).unwrap();
          hp41_core::ops::dispatch(&mut state, Op::Enter).unwrap();
      }
      // After 4 pushes+enters: T=1, Z=2, Y=3, X=4 (HP-41 stack lift behaviour)
      // Actually: last Enter duplicates X=4. Load manually for exactness.
      // Re-push with explicit stack setup:
      let mut state = CalcState::new();
      state.program = mean_prog.ops.clone();
      // Build stack: push 1 ENTER 2 ENTER 3 ENTER 4 → T=1,Z=2,Y=3,X=4
      hp41_core::ops::dispatch(&mut state, Op::PushNum(HpNum::from(1i32))).unwrap();
      hp41_core::ops::dispatch(&mut state, Op::Enter).unwrap();

      let result = f64::from(state.stack.x);
  ```
