    ```rust
    //! Phase 2 storage register operations: STO, RCL, STO+/-/×/÷, CLREG.
    //!
    //! Register addresses are 0–99 (0-indexed). Addresses ≥ 100 return InvalidOp.
    //! STO and STO-arith: Neutral lift (do not modify lift_enabled).
    //! RCL: Enable lift (like PushNum — places a value on the stack).

    use crate::error::HpError;
    use crate::state::CalcState;
    use crate::stack::{apply_lift_effect, enter_number, LiftEffect};
    use crate::ops::StoArithKind;

    /// STO n: copy X register into storage register n. Stack unchanged.
    /// LiftEffect: Neutral. LASTX: not saved (STO is not an arithmetic operation).
    pub fn op_sto(state: &mut CalcState, reg: u8) -> Result<(), HpError> {
        if reg >= 100 {
            return Err(HpError::InvalidOp);
        }
        state.regs[reg as usize] = state.stack.x.clone();
        apply_lift_effect(state, LiftEffect::Neutral);
        Ok(())
    }

    /// RCL n: recall register n into X (with stack lift if lift_enabled).
    /// LiftEffect: Enable. LASTX: not saved.
    pub fn op_rcl(state: &mut CalcState, reg: u8) -> Result<(), HpError> {
        if reg >= 100 {
            return Err(HpError::InvalidOp);
        }
        let val = state.regs[reg as usize].clone();
        // Force lift_enabled = true so enter_number performs the stack lift.
        // This matches HP-41 hardware: RCL always lifts regardless of prior state.
        state.stack.lift_enabled = true;
        enter_number(state, val);
        apply_lift_effect(state, LiftEffect::Enable);
        Ok(())
    }

    /// STO+/−/×/÷ n: apply arithmetic to register n using X.
    /// R[n] ← R[n] OP X. Stack and X are unchanged.
    /// LiftEffect: Neutral. LASTX: not saved.
    ///
    /// IMPORTANT: compute new value FIRST, write ONLY on success (atomicity guarantee).
    pub fn op_sto_arith(state: &mut CalcState, reg: u8, kind: StoArithKind) -> Result<(), HpError> {
        if reg >= 100 {
            return Err(HpError::InvalidOp);
        }
        // Compute first — do NOT write to state.regs[reg] until we know the op succeeds.
        let new_val = match kind {
            StoArithKind::Add => state.regs[reg as usize].checked_add(&state.stack.x)?,
            StoArithKind::Sub => state.regs[reg as usize].checked_sub(&state.stack.x)?,
            StoArithKind::Mul => state.regs[reg as usize].checked_mul(&state.stack.x)?,
            StoArithKind::Div => state.regs[reg as usize].checked_div(&state.stack.x)?,
        };
        // Write only after successful computation (Pitfall 6 guard)
        state.regs[reg as usize] = new_val;
        apply_lift_effect(state, LiftEffect::Neutral);
        Ok(())
    }

    /// CLREG: clear all storage registers to zero.
    /// LiftEffect: Neutral.
    pub fn op_clreg(state: &mut CalcState) -> Result<(), HpError> {
        state.regs = std::array::from_fn(|_| crate::num::HpNum::zero());
        apply_lift_effect(state, LiftEffect::Neutral);
        Ok(())
    }
    ```

    Key implementation notes:
    - FIX n uses Rust's `format!("{:.prec$}", decimal, prec = n)` which handles trailing zeros
    - FIX overflow: if `decimal.abs() >= Decimal::from(10u64.pow(10 - n as u32))` → fall back to SCI 9
    - SCI n: use a format!-based approach:
      1. Find the base-10 exponent: e = floor(log10(|x|))
      2. Compute mantissa = x / 10^e, formatted to n decimal places
      3. Assemble: "{mantissa}E {exp:02}" or "{mantissa}E-{exp:02}"
    - ENG n: same as SCI but e = 3 * floor(e_sci / 3) and mantissa scaled accordingly
    - Special case: zero always uses 0.0000... format appropriate to mode

    ```rust
    //! HP-41 display number formatting.
    //!
    //! Produces HP-41-style display strings for FIX, SCI, and ENG modes.

    use rust_decimal::Decimal;
    use rust_decimal::prelude::ToPrimitive;
    use std::str::FromStr;

    use crate::num::HpNum;
    use crate::state::DisplayMode;

    /// Format an HpNum according to the current display mode.
    /// Returns the HP-41-style display string.
    pub fn format_hpnum(n: &HpNum, mode: &DisplayMode) -> String {
        let d = n.inner();
        match mode {
            DisplayMode::Fix(digits) => format_fix(d, *digits as usize),
            DisplayMode::Sci(digits) => format_sci(d, *digits as usize),
            DisplayMode::Eng(digits) => format_eng(d, *digits as usize),
        }
    }

    /// Format the ALPHA register for display.
    /// HP-41 display is 12 characters; truncate if longer.
    pub fn format_alpha(reg: &str) -> String {
        reg.chars().take(12).collect()
    }

    // ── FIX formatting ────────────────────────────────────────────────────────

    fn format_fix(d: Decimal, digits: usize) -> String {
        if d.is_zero() {
            if digits == 0 {
                return "0.".to_string();
            }
            return format!("0.{:0>width$}", "", width = digits);
        }

        // FIX overflow check: if |integer part| requires more than (10 - digits) characters,
        // fall back to SCI 9 display (HP-41 hardware behavior).
        let abs_d = d.abs();
        let overflow_threshold = Decimal::from_str(
            &format!("1E{}", 10_usize.saturating_sub(digits))
        ).unwrap_or(Decimal::MAX);

        if abs_d >= overflow_threshold {
            return format_sci(d, 9);
        }

        // Rust's format! with precision handles FIX formatting including trailing zeros.
        let s = format!("{:.prec$}", d, prec = digits);

        // HP-41 FIX 0 shows trailing decimal point: "42."
        if digits == 0 {
            format!("{s}.")
        } else {
            s
        }
    }

    // ── SCI formatting ────────────────────────────────────────────────────────

    fn format_sci(d: Decimal, digits: usize) -> String {
        if d.is_zero() {
            let zeros = if digits > 0 {
                format!(".{:0>width$}", "", width = digits)
            } else {
                ".".to_string()
            };
            return format!("0{zeros}E 00");
        }

        let (mantissa_str, exp) = decompose_sci(d, digits);
        format_sci_assembled(&mantissa_str, exp)
    }

    /// Decompose a Decimal into (formatted_mantissa_string, exponent) for SCI display.
    fn decompose_sci(d: Decimal, digits: usize) -> (String, i32) {
        let is_negative = d.is_sign_negative();
        let abs_d = d.abs();

        // Compute base-10 exponent
        let exp = if abs_d.is_zero() {
            0i32
        } else {
            // floor(log10(|d|))
            let log = abs_d.to_f64().unwrap_or(1.0).log10().floor() as i32;
            log
        };

        // Compute mantissa = d / 10^exp
        let ten_to_exp = if exp >= 0 {
            Decimal::from_str(&format!("1E{exp}")).unwrap_or(Decimal::ONE)
        } else {
            Decimal::from_str(&format!("1E{exp}")).unwrap_or(Decimal::ONE)
        };

        let mantissa = if let Some(m) = abs_d.checked_div(ten_to_exp) {
            m
        } else {
            abs_d
        };

        let mantissa_formatted = format!("{:.prec$}", mantissa, prec = digits);
        let sign = if is_negative { "-" } else { "" };
        (format!("{sign}{mantissa_formatted}"), exp)
    }

    /// Assemble a SCI string from mantissa string and exponent.
    fn format_sci_assembled(mantissa: &str, exp: i32) -> String {
        if exp >= 0 {
            format!("{mantissa}E {:02}", exp)
        } else {
            format!("{mantissa}E-{:02}", -exp)
        }
    }

    // ── ENG formatting ────────────────────────────────────────────────────────

    fn format_eng(d: Decimal, digits: usize) -> String {
        if d.is_zero() {
            let zeros = if digits > 0 {
                format!(".{:0>width$}", "", width = digits)
            } else {
                ".".to_string()
            };
            return format!("0{zeros}E 00");
        }

        let is_negative = d.is_sign_negative();
        let abs_d = d.abs();

        // Compute SCI exponent
        let sci_exp = if abs_d.is_zero() {
            0i32
        } else {
            abs_d.to_f64().unwrap_or(1.0).log10().floor() as i32
        };

        // Clamp to multiple of 3 (round DOWN, i.e., towards negative infinity)
        let eng_exp = if sci_exp >= 0 {
            (sci_exp / 3) * 3
        } else {
            // For negative exponents: e.g., sci_exp=-1 → eng_exp=-3, sci_exp=-4 → eng_exp=-6
            ((sci_exp - 2) / 3) * 3
        };

        // Mantissa = d / 10^eng_exp; will have 1–3 digits before the decimal point
        let ten_to_eng_exp = Decimal::from_str(&format!("1E{eng_exp}")).unwrap_or(Decimal::ONE);
        let mantissa = if let Some(m) = abs_d.checked_div(ten_to_eng_exp) {
            m
        } else {
            abs_d
        };

        let mantissa_formatted = format!("{:.prec$}", mantissa, prec = digits);
        let sign = if is_negative { "-" } else { "" };
        let mantissa_str = format!("{sign}{mantissa_formatted}");

        format_sci_assembled(&mantissa_str, eng_exp)
    }
    ```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-13 | Tampering | StoReg/RclReg out-of-range address | mitigate | `if reg >= 100 { return Err(HpError::InvalidOp) }` before ANY array access in all 3 bounded ops |
| T-02-14 | Tampering | StoArith partial write on overflow | mitigate | Compute `new_val = ...?` first; only write `state.regs[reg] = new_val` after success |
