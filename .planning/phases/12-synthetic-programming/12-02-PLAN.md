```rust
/// Map a crossterm KeyCode to the HP-41 hardware key code (row×10 + col, 1-indexed).
/// Returns 0 for keys with no HP-41 hardware equivalent (function keys, Ctrl combos, etc.).
/// Called from `App::handle_key()` to update `CalcState.last_key_code` on every Press
/// event (D-01). Read by `Op::GetKey` to push the last key code to X (SYNT-01).
///
/// HP-41C keyboard layout: 8 rows × 5 columns. Key code = row × 10 + col.
/// Rows are numbered 1-8 top-to-bottom, columns 1-5 left-to-right.
/// (Source: HP-41C Owner's Manual Appendix A; rows 1-4 marked [ASSUMED] in RESEARCH.md A1.)
pub fn keycode_to_hp41_code(code: crossterm::event::KeyCode) -> u8 {
    use crossterm::event::KeyCode;
    match code {
        // Row 8: 0, ., EEX, ENTER (bottom row)
        KeyCode::Char('0') => 81,
        KeyCode::Char('.') => 82,
        KeyCode::Char('e') => 83, // EEX
        KeyCode::Enter => 84,     // ENTER
        // Row 7: 1, 2, 3, +
        KeyCode::Char('1') => 71,
        KeyCode::Char('2') => 72,
        KeyCode::Char('3') => 73,
        KeyCode::Char('+') => 74,
        // Row 6: 4, 5, 6, -
        KeyCode::Char('4') => 61,
        KeyCode::Char('5') => 62,
        KeyCode::Char('6') => 63,
        KeyCode::Char('-') => 64,
        // Row 5: 7, 8, 9, *
        KeyCode::Char('7') => 51,
        KeyCode::Char('8') => 52,
        KeyCode::Char('9') => 53,
        KeyCode::Char('*') => 54,
        // Row 4: USER, f-shift, g-shift, ENTER (alternate), ÷
        KeyCode::Char('u') | KeyCode::Char('U') => 41, // USER
        KeyCode::Char('f') => 42,                       // f-key (lowercase: format cycle)
        KeyCode::Char('g') => 43,                       // g-key (lowercase: CLREG)
        KeyCode::Char('/') => 45,                       // ÷
        // Row 3: R/S, SST, BST/GTO, COS, TAN
        KeyCode::F(5) => 31,                            // R/S
        KeyCode::F(7) => 32,                            // SST
        KeyCode::F(8) => 32,                            // BST (treated as SST for code purposes)
        KeyCode::Char('C') => 34,                       // COS (uppercase)
        KeyCode::Char('T') => 35,                       // TAN (uppercase)
        // Row 2: XEQ, STO, RCL, R↓, SIN
        KeyCode::Char('X') => 21, // XEQ key code (uppercase X also opens HexModal — see app.rs)
        KeyCode::Char('S') => 22, // STO modal opener — key code is still 22 (STO key)
        KeyCode::Char('R') => 23, // RCL modal opener — key code is still 23 (RCL key)
        KeyCode::Char('r') => 24, // R↓ (lowercase r)
        KeyCode::Char('q') => 25, // SIN (Phase 8 reassignment to 'q')
        // Row 1: Σ+, 1/x, √x, LOG, LN
        KeyCode::Char('z') => 11, // Σ+
        KeyCode::Char('I') => 12, // 1/x
        KeyCode::Char('s') => 13, // √x (lowercase s)
        KeyCode::Char('L') => 15, // LN (uppercase L)
        // All other keys (Esc, F1-F4, Backspace, Tab, arrow keys, modifiers, unmapped chars):
        // no HP-41 hardware equivalent → 0.
        _ => 0,
    }
}
```

                    self.pending_input = None;

                    // matches PrintModal arm fallthrough).
                    self.pending_input = Some(PendingInput::HexModal(acc.clone()));
                }
            }
        }
