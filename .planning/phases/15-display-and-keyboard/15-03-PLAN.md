```typescript
function resolveKeyId(e: KeyboardEvent, state: CalcStateView | null): string | null {
  // EEX-CHS: 'n' routes to "eex_chs" when in EEX entry mode (D-06)
  if (e.key === 'n') return state?.in_eex_mode ? 'eex_chs' : 'chs';
  // Digit entry — key_id is the digit character itself
  if (e.key.length === 1 && '0123456789'.includes(e.key)) return e.key;
  if (e.key === '.') return '.';
  if (e.key === 'e') return 'e';
  // Modal-trigger keys — silently ignore (D-05): no invoke, no error
  if (e.key.length === 1 && 'SRfFPX'.includes(e.key)) return null;
  // Named op mapping — source: hp41-cli/src/keys.rs key_to_op()
  const MAP: Record<string, string> = {
    'Enter': 'enter', 'Backspace': 'clx',
    '+': 'plus', '-': 'minus', '*': 'mul', '/': 'div',
    'r': 'rdn', 'x': 'xy_swap', 'l': 'lastx', 's': 'sqrt', 'p': 'prgm_mode',
    'a': 'asin', 'c': 'acos', 'k': 'atan',
    'C': 'cos', 'T': 'tan', 'L': 'ln', 'G': 'log', 'E': 'exp',
    'H': 'tenpow', 'I': 'recip', 'W': 'sq', 'Y': 'ypow',
    'u': 'user_mode',
    'z': 'sigma_plus', 'Z': 'sigma_minus', 'm': 'mean', 'D': 'sdev',
    'y': 'yhat', 'b': 'lr', 'O': 'corr', 'V': 'cl_sigma_stat',
    'h': 'hms_to_h', 'j': 'hms_add', 'J': 'hms_sub',
    'q': 'sin',    // Phase 8 binding: 'q' = SIN
    'g': 'clreg',  // Phase 8 binding: 'g' = CLREG
  };
  return MAP[e.key] ?? null;
}
```
e.key for shifted characters gives the capital letter directly (e.g., Shift+C → 'C').
No modifier check needed — matches crossterm KeyCode::Char('C') convention from keys.rs.

React hook pattern (D-07/D-12 — mandatory StrictMode-safe keyboard listener):
```typescript
const busyRef = useRef(false);  // useRef, NOT useState — changes must NOT trigger re-renders

const handleKey = useCallback((e: KeyboardEvent) => {
  if (busyRef.current) return;              // debounce: ignore while invoke pending
  const keyId = resolveKeyId(e, calcState);
  if (keyId === null) return;
  e.preventDefault();
  busyRef.current = true;
  invoke<CalcStateView>('dispatch_op', { keyId })
    .then(view => setCalcState(view))
    .catch(err => console.error('dispatch_op error:', err))
    .finally(() => { busyRef.current = false; });
}, [calcState]); // calcState in deps: resolveKeyId reads in_eex_mode from latest state

useEffect(() => {
  window.addEventListener('keydown', handleKey);
  return () => window.removeEventListener('keydown', handleKey); // REQUIRED: cleanup for StrictMode
}, [handleKey]);
```

    ```typescript
    import { invoke } from '@tauri-apps/api/core';

      user: boolean;
      prgm: boolean;
      alpha: boolean;
      rad: boolean;
      grad: boolean;
    }

    interface CalcStateView {
      display_str: string;
      x_str: string;
      y_str: string;
      z_str: string;
      t_str: string;
      lastx_str: string;
      in_eex_mode: boolean;
      annunciators: Annunciators;
      print_lines: string[];

      // EEX-CHS: 'n' routes based on current in_eex_mode (D-06)
      if (e.key === 'n') return state?.in_eex_mode ? 'eex_chs' : 'chs';
      // Digit entry
      if (e.key.length === 1 && '0123456789'.includes(e.key)) return e.key;
      if (e.key === '.') return '.';
      if (e.key === 'e') return 'e';
      // Modal-trigger keys — silently ignore, no invoke (D-05)
      if (e.key.length === 1 && 'SRfFPX'.includes(e.key)) return null;
      // Named op mapping — authoritative source: hp41-cli/src/keys.rs key_to_op()
      const MAP: Record<string, string> = {
        'Enter': 'enter', 'Backspace': 'clx',
        '+': 'plus', '-': 'minus', '*': 'mul', '/': 'div',
        'r': 'rdn', 'x': 'xy_swap', 'l': 'lastx', 's': 'sqrt', 'p': 'prgm_mode',
        'a': 'asin', 'c': 'acos', 'k': 'atan',
        'C': 'cos', 'T': 'tan', 'L': 'ln', 'G': 'log', 'E': 'exp',
        'H': 'tenpow', 'I': 'recip', 'W': 'sq', 'Y': 'ypow',
        'u': 'user_mode',
        'z': 'sigma_plus', 'Z': 'sigma_minus', 'm': 'mean', 'D': 'sdev',
        'y': 'yhat', 'b': 'lr', 'O': 'corr', 'V': 'cl_sigma_stat',
        'h': 'hms_to_h', 'j': 'hms_add', 'J': 'hms_sub',
        'q': 'sin',    // Phase 8 reassignment: 'q' = SIN
        'g': 'clreg',  // Phase 8 addition: 'g' = CLREG
      };
      return MAP[e.key] ?? null;
    }

    function App() {
      const [calcState, setCalcState] = useState<CalcStateView | null>(null);
      const busyRef = useRef(false);

      // Mount: load initial state via get_state (D-11 — no polling)
      useEffect(() => {
        invoke<CalcStateView>('get_state')
          .then(view => setCalcState(view))
          .catch(err => console.error('get_state error:', err));
      }, []);

      // Keyboard handler — useCallback with calcState dep so 'n' reads latest in_eex_mode
      const handleKey = useCallback((e: KeyboardEvent) => {

      useEffect(() => {
        window.addEventListener('keydown', handleKey);
        return () => window.removeEventListener('keydown', handleKey);
      }, [handleKey]);

      if (!calcState) {
        return <div className="calculator"><div className="display">Loading...</div></div>;
      }

      const annunciatorNames = ['user', 'prgm', 'alpha', 'rad', 'grad'] as const;
      const stackRows: [string, string][] = [
        ['X', calcState.x_str],
        ['Y', calcState.y_str],
        ['Z', calcState.z_str],
        ['T', calcState.t_str],
        ['L', calcState.lastx_str],
      ];

      return (
        <div className="calculator">
          <div className="annunciators">
            {annunciatorNames.map(name => (
              <span
                key={name}
                className={`annunciator${calcState.annunciators[name] ? ' active' : ''}`}
              >
                {name.toUpperCase()}
              </span>
            ))}
          </div>
          <div className="display">{calcState.display_str}</div>
          <div className="stack-panel">
            {stackRows.map(([label, value]) => (
              <div key={label} className="stack-row">
                <span className="stack-label">{label}:</span>
                <span>{value}</span>
              </div>
            ))}
          </div>
          <div id="keyboard-area" />
        </div>
      );
    }

    export default App;
    ```

| Boundary | Description |
|----------|-------------|
| Physical keyboard → React event handler | User keydown events; resolveKeyId filters to a hard-coded allowlist of key IDs |
| React → Tauri IPC | invoke sends a string key_id; Rust validates via key_map::resolve() returning Err for unknowns |
| CalcStateView → JSX render | display_str and register values rendered as text nodes — safe JSX default |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-15-03-01 | Tampering | resolveKeyId — arbitrary key_id injection | mitigate | resolveKeyId returns only strings from a hard-coded MAP or null; Rust key_map::resolve() returns GuiError for unknowns — no panic |
| T-15-03-02 | Denial of Service | busyRef — rapid keypress flooding invoke() | mitigate | busyRef.current guard in handleKey; only one invoke() in-flight at a time per D-07 |
| T-15-03-03 | Tampering | display_str rendered as text content | accept | JSX renders text nodes by default (no inner HTML); HP-41 display strings contain only ASCII — no injection surface |
| T-15-03-04 | Tampering | StrictMode double-mount registers two listeners | mitigate | useEffect cleanup return calls removeEventListener; confirmed by SC-4 in human-verify checkpoint |
</threat_model>
