**Component structure (D-08, D-11, D-12, D-13):**

```typescript
import { useState } from 'react';

type KeyDef = {
  id: string;
  label: string;
  fShiftLabel?: string;
  row: number;
  col: number;
  colSpan?: number;
};

const KEY_W = 39;
const KEY_H = 26;
const GAP = 4;
const PAD = 8;
const FSHIFT_H = 12;

// [KEY_DEFS array here — 40 entries as specified above]

function getKeyColor(key: KeyDef): string {
  if (key.row === 0) return '#3a2a1e';
  if (key.id === 'enter') return '#1a3a1a';
  if (['user_mode', 'prgm_mode', 'alpha_toggle'].includes(key.id) ||
      (key.id === '' && (key.label === 'f' || key.label === 'g'))) return '#d4c9b0';
  return '#1a1a1a';
}

interface KeyboardProps {
  onKey: (keyId: string) => void;
  busyRef: React.MutableRefObject<boolean>;
}

export function Keyboard({ onKey, busyRef }: KeyboardProps) {
  const [pressedKey, setPressedKey] = useState<string | null>(null);

  const handleKeyClick = (keyId: string) => {
    if (!keyId) return;                    // visual-only key guard (D-07, Pitfall 2)
    if (busyRef.current) return;          // debounce guard (D-13)
    setPressedKey(keyId);
    setTimeout(() => setPressedKey(prev => prev === keyId ? null : prev), 150);
    onKey(keyId);
  };

  return (
    <svg
      width="100%"
      viewBox="0 0 400 230"
      xmlns="http://www.w3.org/2000/svg"
      aria-label="HP-41C keyboard"
    >
      <rect width="400" height="230" fill="#3d2b1f" rx={6} />
      {KEY_DEFS.map(key => {
        const cs = key.colSpan ?? 1;
        const x = PAD + key.col * (KEY_W + GAP);
        const rectY = PAD + key.row * (FSHIFT_H + KEY_H + GAP) + FSHIFT_H;
        const w = cs * KEY_W + (cs - 1) * GAP;
        const isPressed = pressedKey === key.id && Boolean(key.id);
        const labelColor = ['user_mode', 'prgm_mode', 'alpha_toggle'].includes(key.id) ||
                           (key.id === '' && (key.label === 'f' || key.label === 'g'))
                           ? '#2a1a0a' : '#ffffff';
        return (
          <g
            key={`${key.row}-${key.col}`}
            onClick={() => handleKeyClick(key.id)}
            className={isPressed ? 'key key-pressed' : 'key'}
            style={{ pointerEvents: 'all' }}
          >
            {key.fShiftLabel && (
              <text
                x={x + w / 2}
                y={rectY - 2}
                textAnchor="middle"
                fill="#c8a400"
                fontSize={8}
              >
                {key.fShiftLabel}
              </text>
            )}
            <rect
              x={x}
              y={rectY}
              width={w}
              height={KEY_H}
              rx={3}
              ry={3}
              fill={getKeyColor(key)}
              stroke="#111"
              strokeWidth={0.5}
            />
            <text
              x={x + w / 2}
              y={rectY + KEY_H / 2 + 4}
              textAnchor="middle"
              fill={labelColor}
              fontSize={10}
              fontWeight="bold"
            >
              {key.label}
            </text>
          </g>
        );
      })}
    </svg>
  );
}
```

**Critical implementation notes:**
- The `KEY_DEFS` array must be declared `as const` at the end: `] as const;` — use `const KEY_DEFS: KeyDef[] = [...]` (not `as const` on the array itself since readonly fields conflict with `KeyDef[]`). Actually use just `const KEY_DEFS: KeyDef[] = [` without `as const` since fShiftLabel optional field conflicts with `as const` readonly string narrowing.
- Use `key={``${key.row}-${key.col}``}` on the `<g>` element (not `key.id`) because multiple keys have `id: ''` and React requires unique keys.
- The `import React` style is NOT needed — using JSX transform (Vite with @vitejs/plugin-react). Just `import { useState } from 'react'`.
- `React.MutableRefObject<boolean>` type: since we only import `useState`, we need `import React from 'react'` OR use the inline form `import { useState, type MutableRefObject } from 'react'` and type as `MutableRefObject<boolean>`. Use the latter to avoid a default React import.
