fn test_sdev_sample_two_points() {
    // Data: (X=2,Y=4) and (X=4,Y=8)
    // n=2, Σx=6, Σx²=20, Σy=12, Σy²=80
    // σx = sqrt((2·20 − 36) / (2·1)) = sqrt(4/2) = sqrt(2) ≈ 1.4142...
    // σy = sqrt((2·80 − 144) / (2·1)) = sqrt(16/2) = sqrt(8) ≈ 2.8284...
    let mut s = CalcState::new();
    set_y(&mut s, 4);
    push(&mut s, 2);
    dispatch(&mut s, Op::SigmaPlus).unwrap();
    set_y(&mut s, 8);
    push(&mut s, 4);
    dispatch(&mut s, Op::SigmaPlus).unwrap();

    dispatch(&mut s, Op::Sdev).unwrap();
    // σx = sqrt(2) — verify it's positive and approximately correct
    assert!(s.stack.x.inner() > Decimal::ZERO, "σx must be positive");
    // σy = 2*σx (linear relationship Y=2X in our data set)
    let two_sx = s.stack.x.checked_mul(&HpNum::from(2i32)).unwrap();
    assert_eq!(s.stack.y.inner(), two_sx.inner(), "σy must be 2·σx for Y=2X data");
}
